## Usage Example

```python
# Local model runner
from skill_local_llama_runner import LocalLlamaRunner
runner = LocalLlamaRunner(profile="fast")

# Verify health
if runner.is_healthy():
    result = runner.complete(
        prompt="Summarize...",
        max_tokens=500
    )
    print(f"Saved: {result.completion_tokens} tokens")

# Overnight batch
from skill_overnight_batch_engine import BatchEngine, JobConfig
engine = BatchEngine()
engine.queue_job(JobConfig(
    name="newsletter_digest",
    skill="skill-newsletter-processor",
    model_profile="overnight"
))
results = engine.run_batch()

# Structured output
from skill_structured_output_guardian import OutputGuardian
from pydantic import BaseModel

class Summary(BaseModel):
    title: str
    points: list[str]

guardian = OutputGuardian()
result = guardian.generate_and_validate(
    prompt="Summarize this...",
    schema=Summary
)
```
